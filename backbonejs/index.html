Welcome Backbone.js

  • Created by Jeremy Ashkenas (DocumentCloud) who built CoffeeScript and underscore.js
  • Super small and lightweight - 3.9kb compressed
  • It provides structure to your js apps with MVC
  • Backend agnostic

Model

A model is piece of data.

	
Tour = {
	"name": "Lovely Tour",
	"price": "15",
    	"pid": 0
}

How you set up a backbone model

var Tour = Backbone.Model.extend({  
       initialize: function(){  
           alert("Sup! ");  
       }
 });

Collection

A collection is collection of Models

	
TourCollection = [
		{
	        "name": "Lovely Tour",
	        "price": "15",
			"pid": 0
	    	},
		{
		    "name": "Awesome Tour",
		    "price": "15",
			"pid": 1
		},
]

Create a backbone collection

TourCollection = Backbone.Collection.extend({
	//tie the collection to a model
	model: Tour,

	//If you define a comparator, it will be used to maintain 
	//the collection in sorted order.
	comparator: function(item) {
	       return item.get('pid');
	   }
});

Views

  • Views listen to events thrown by DOM or even changes in collections/model
  • Views are tied with collections or models

Instantiating a View

		new TourListView({model:tours});

Backbone Views

var TourListView = Backbone.View.extend({
   //associate dom element with this view
   el: $('#tourListView'),
   events: {
		"click .contact" : "contactBtnClick",
		"hover .picture" : "pictureHover",
   },
   render: function() {
       //show this view
   },
   contactBtnClick: function() {},
   pictureHover: function() {}
});
